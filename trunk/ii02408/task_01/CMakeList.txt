cmake_minimum_required(VERSION 3.13) # Проверка версии CMake
project(task_01) # Создание проекта "task_01"

# Добавление исходного файла main.cpp в папку src
set(SOURCE_FILES src/main.cpp)

# Добавление папки doc в проект
add_subdirectory(doc)

# Создание исполняемой цели с исходными файлами, указанными в переменной SOURCE_FILES
add_executable(task_01 ${SOURCE_FILES})

# Линковка с библиотеками из папки doc
target_link_libraries(task_01 doc)
